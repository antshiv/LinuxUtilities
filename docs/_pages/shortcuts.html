Shell Helpers

# presenter_dash.sh commands
./presenter_dash.sh anchor   # set anchor at cursor position
./presenter_dash.sh dash     # dashed line from anchor → cursor
./presenter_dash.sh dot      # dotted line
./presenter_dash.sh solid    # solid line
./presenter_dash.sh arrow    # arrow line
./presenter_dash.sh clear    # clear all
./presenter_dash.sh undo     # undo last stroke
./presenter_dash.sh redo     # redo stroke
./presenter_dash.sh reset    # remove anchor
